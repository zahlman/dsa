# Preferred format when disassembling: a single value.
# The output base and signedness may be configured.
# Assembles to the same *numeric value* regardless of platform endianness.

32:value base:<base> signed:<signed>
	:

----

# Any base is accepted when assembling, so no base: flags here.
# When assembling from components, always expect unsigned values.

16:m
    :
16:n
    :

----

# Similarly when assembling from 4 components.
# Assembles to the same *sequence of bytes* regardless of platform endianness.

8:w
    :
8:x
    :
8:y
    :
8:z
    :
